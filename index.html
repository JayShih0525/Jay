self.grid_size) if row < self.grid_size - 1: # Down neighbors.append(index + self.grid_size) if col > 0: # Left neighbors.append(index - 1) if col < self.grid_size - 1: # Right neighbors.append(index + 1) return neighbors @staticmethod def heuristic(state): """Manhattan distance heuristic.""" distance = 0 for i, value in enumerate(state): if value == 0: continue goal_row, goal_col = divmod(value -
